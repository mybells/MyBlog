# 11. 环中最长子串
![11](/images/od/11.png)

```js
const readline = require('readline');
const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
});

let inputs = [];
rl.on('line', (input) => {
    const len = input.length;
    let num = 0;
    for(let v of input) {
        if(v === '0') {
            num++;
        }
    }
    if (num % 2 === 0) {
        console.log(num);
    } else {
        console.log(len - 1);
    }
});

```
