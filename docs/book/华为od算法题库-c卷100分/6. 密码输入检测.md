# 6. 密码输入检测
![6](/images/od/6.png)

```js
const readline = require('readline');
const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
});

rl.on('line', (input) => {
    let res = '';
    let isBig = false;
    let isSmall = false;
    let isNum = false;
    let isSpec = false;
    for(let i=0; i<input.length; i++) {
        if (input[i] === '<') {
            res = res.slice(0, -1);
        } else {
            res += input[i];
        }
    }
    for (let i=0; i<res.length; i++) {
        if (/[0-9]/.test(res[i])) {
            isNum = true;
        } else if (/[a-z]/.test(res[i])) {
            isSmall = true;
        } else if (/[A-Z]/.test(res[i])) {
            isBig = true;
        } else if (res[i]) {
            isSpec = true;
        }
    }
    let flag = res.length >= 8 && isNum && isSmall && isBig && isSpec;
    console.log(res + ',' + flag);
})

```
