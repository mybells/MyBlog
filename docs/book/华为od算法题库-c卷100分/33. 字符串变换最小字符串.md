# 33. 字符串变换最小字符串

```js
const readline = require('readline');
const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
});

rl.on('line', function(line) {
    const sortedArr = s.split('').sort();
    if (sortedArr.join('') === s) {
        console.log(s);
        return;
    }

    let sb = s.split('');
    for(let i=0; i<s.length; i++) {
        if (s[i] !== sortedArr[i]) {
            let swapIndex = -1;
            for(let j = i+1; j<s.length; j++) {
                if (sb[j] === sortedArr[i]) {
                    swapIndex = j;
                }
            }
            [sb[i], sb[swapIndex]] = [sb[swapIndex], sb[i]];
            break;
        }
    }

    console.log(sb.join(''));
})
```
