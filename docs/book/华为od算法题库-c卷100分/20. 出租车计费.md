# 20. 出租车计费
![20](/images/od/20.png)

```js
const readline = require('readline');
const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
});

rl.on('line', function(str)) {
    let ans = 0;
    for(let i=0; i<str.length; i++) {
        let dight = parseInt(str[i]);
        if (dight > 4) {
            dight--;
        }
        ans = ans * 9 + dight;
    }
    console.log(ans);
}

```
