# 51. 寻找连续区间-数组连续和
![51](/images/od/51.png)

```js
const readline = require('readline');
const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
});
let n = 0;
let x = 0;
let nums = [];
rl.on('line', function(line) {
    if (!n) {
        [n, x] = line.split(' ').map(Number);
    } else {
        nums = line.split(' ').map(Number);
        let left = 0;
        let right = 0;
        let count = 0;
        let sum = 0;

        while(right < n) {
            sum += nums[right];
            while(sum >= x) {
                count += n - right;
                sum -= nums[left];
                left++;
            }
            right++;
        }
        console.log(count);
    }
});
```
