```js
const readline = require('readline');
const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
});
let input = [];
rl.on('line', function(num) {
    let [k, n, m] = num.split(' ').map(Number);
    if (k < 0 || n > 0 || m <= 1 || n >=m) {
        console.log(0);
    } else {
        let ans = k.toString(m);
        let count = 0;
        for(let i=0; i<ans.length; i++) {
            if (ans[i] == n) {
                count++;
            }
        }
        console.log(count);
    } 
});
```
