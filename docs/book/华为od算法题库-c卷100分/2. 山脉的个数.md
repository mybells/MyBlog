```js
function count_peaks(a) {
    let count = 0;
    for(let i=0; i<a.length; i++) {
        if (i===0 && a[i] > a[i+1]) {
            count++;
        } 
        if (i===a.length-1 && a[i] > a[i-1]) {
            count++;
        }
        if (i>0 && i<a.length-1 && a[i] > a[i-1] && a[i] > a[i+1]) {
            count++;
        }
    }
    return count;
}
```
